f=function(x){
  y=((4*x)/((x*x)+1))+3
  return (y)
}

g=function(x){
  y=(x*x)-(1.2*x)+3
  return (y)
}

n = 10000

#prenons n points au hasard dans [0;2]*[2.5;5]
E=matrix(c(runif(n, 0, 2),runif(n = n, 2.5 , 5)) , ncol = 2 , nrow = n)

Y=c(1:n)


for (i in 1:n)
  {
  Y[i] = (E[i,2]<f(E[i,1]))&(E[i,2]>g(E[i,1]))
}
